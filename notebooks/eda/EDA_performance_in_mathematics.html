predictors  <- clean_data[,-ncol(clean_data)]
predictors_names <- names(predictors)
predictors_names = set_names(predictors_names)
par(mfrow = c(5,6))
hist_fun <-  function(x) {
     ggplot(clean_data, aes(x = .data[[x]])) +
          geom_histogram(stat = "count", bins = 5)  +
          theme_bw() +
          theme(axis.text=element_text(size = 15),
                axis.title=element_text(size=17,face="bold"))
}

plots <- map(predictors_names, ~hist_fun(.x) ) 

cowplot::plot_grid(plotlist = plots, nrow = 6)
